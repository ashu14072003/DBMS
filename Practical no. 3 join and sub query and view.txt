CREATE TABLE employees (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NAME VARCHAR(50) NOT NULL,
    AGE INT,
    GENDER VARCHAR(10),
    CITY VARCHAR(50),
    FOREIGN KEY (CITY) REFERENCES cities(City)
);
